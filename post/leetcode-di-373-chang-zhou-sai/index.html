100139. 循环移位后的矩阵相似检查

中文题面不翻译下标从哪开始😅

class Solution {
public:
    bool areSimilar(vector<vector<int>>& mat, int k) {
        int n = mat.size(), m = mat[0].size();
        k %= m;
        for (int i = 0;i < n;i++){
            if (i%2 == 0){
                for (int j = 0;j < m;j++){
                    if (mat[i][(j+k)%m] != mat[i][j]) return false;
                }
            }else{
                for (int j = 0;j < m;j++){
                    if (mat[i][(j-k+m)%m] != mat[i][j]) return false;
                }
            }
        }
        return true;
    }
};

100134. 统计美丽子字符串 I

class Solution {
public:
    int beautifulSubstrings(string s, int k) {
        int n = s.size(), ans = 0;
        for (int i = 0;i < n;i++){
            int x = 0, y = 0;
            for (int j = i;j < n;j++){
                if (s[j]=='a'||s[j]=='e'||s[j]=='i'||s[j]=='o'||s[j]=='u'){
                    x++;
                }else{
                    y++;
                }
                if (x == y && (x*y)%k == 0) ans++;
            }
        }
        return ans;
    }
};

100142. 交换得到字典序最小的数组

把每个数看成一个点，如果两个数可以互换，则两个数就有一条无向边。

这样就会形成一个连通图：连通图里的任何数都可以互换。

  1. 将数组从小到大排序，排序时要保留原始下标。
  2. 从小到大遍历，依次获得一个连通图。
  3. 将连通图中所有数对应的原始下标从小到大排序。
  4. 将将连通图中所有数从小到大，填入排好序的下标中。
class Solution {
public:
    vector<int> lexicographicallySmallestArray(vector<int>& nums, int limit) {
        int n = nums.size();
        vector<vector<int>> tp(n);
        for (int i = 0;i < n;i++) tp[i] = {nums[i], i};
        sort(tp.begin(), tp.end(), [&](vector<int>& x, vector<int>& y){
            return x[0] < y[0];
        });
        vector<int> ans(n, 0);
        for (int i = 0;i < n;){
            vector<int> dx = {tp[i][1]};
            int j = i+1;
            while (j < n && tp[j][0]-tp[j-1][0] <= limit){
                dx.push_back(tp[j][1]);
                j++;
            }
            sort(dx.begin(), dx.end());
            for (int k = i;k < j;k++) ans[dx[k-i]] = tp[k][0];
            i = j;
        }
        return ans;
    }
};
